macvlan interface for bypassing a VPN connection

  1. ip l a l [exisiting_interface] name [virtual_interface] type macvlan
  2. ip netns a [namespace]
  3. ip l s [virtual_interface] netns [namespace]
  4. ip netns exec [namespace] ip l s [virtual_interface] up
  5. ip netns exec [namespace] dhclient [virtual_interface]
  6. mkdir -p /etc/netns/[namespace]
  7. edit /etc/netns/[namespace]/resolv.conf
  8. ip netns exec [namespace] su --command='[command]' [user]
